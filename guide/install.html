# 安装及使用

# 安装

npm install --save vue-sms-codes

# 全局引用


import smsCode from 'vue-sms-codes'
Vue.use(smsCode)

# 按需引入


import smsCode from 'vue-sms-codes'

...

components: {
  'sms-code': smsCode
}
